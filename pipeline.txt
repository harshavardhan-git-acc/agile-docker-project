pipeline {
 agent any
 environment {
 DOCKER_IMAGE = " harshavardhandwaraka/java-app:latest"
 }
 stages {
 stage('Clone Repository') {
 steps {
 git branch: 'main',
 url: 'https://github.com/harshavardhan-git-acc/agile-docker-project.git'
 }
 }
 stage('Build with Maven') {
 steps {
 bat 'mvn clean package'
 }
 }
 stage('Build Docker Image') {
 steps {
 bat "docker build -t %DOCKER_IMAGE% ."
 }
 }
 stage('Push to DockerHub') {
 steps {
 withCredentials([usernamePassword(
 credentialsId: 'dockerhub-creds',
 usernameVariable: 'harshavardhandwaraka',
 passwordVariable: 'shFNJD@123'
 )]) {
 bat """
 docker login -u harshavardhandwaraka -p shFNJD@123
 docker push %DOCKER_IMAGE%
 """
 }
 }
 }
 stage('Deploy to Kubernetes') {
 steps {
 bat 'kubectl apply -f deployment.yaml'
 }
 }
 }
 post {
 success {
 echo 'CI/CD Pipeline executed successfully!'
 }
 failure {
 echo 'Pipeline failed. Please check logs.'
 }
 }
}
